#!/usr/bin/env bash
set -euo pipefail

PASS=0
FAIL=0
SKIP=0
NAMESPACE="firecrawl"
BASE_DIR="${BASE_DIR:-k8s/base}"

check_manifest() {
  local id="$1" desc="$2" file="$3" pattern="$4"
  printf "%-8s %-60s " "$id" "$desc"
  if [ -f "$file" ] && grep -q "$pattern" "$file"; then
    echo "PASS"
    ((PASS++))
  elif [ ! -f "$file" ]; then
    echo "SKIP (file not found: $file)"
    ((SKIP++))
  else
    echo "FAIL"
    ((FAIL++))
  fi
}

check_manifest_absent() {
  local id="$1" desc="$2" file="$3" pattern="$4"
  printf "%-8s %-60s " "$id" "$desc"
  if [ -f "$file" ] && ! grep -q "$pattern" "$file"; then
    echo "PASS"
    ((PASS++))
  elif [ ! -f "$file" ]; then
    echo "SKIP (file not found: $file)"
    ((SKIP++))
  else
    echo "FAIL (found forbidden pattern: $pattern)"
    ((FAIL++))
  fi
}

check_cluster() {
  local id="$1" desc="$2" cmd="$3"
  printf "%-8s %-60s " "$id" "$desc"
  if ! command -v kubectl &>/dev/null; then
    echo "SKIP (kubectl not available)"
    ((SKIP++))
    return
  fi
  if eval "$cmd" >/dev/null 2>&1; then
    echo "PASS"
    ((PASS++))
  else
    echo "FAIL"
    ((FAIL++))
  fi
}

- Exit code 1 on any FAIL for CI integration
